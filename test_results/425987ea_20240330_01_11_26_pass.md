def min_moves(var_0):
    dp = [0, 1] + [0] * (var_0 - 1)
    for i in range(2, var_0 + 1):
        dp[i] = i
        for x in range(1, i):
            broken = x - 1

            dp[i] = min(dp[i], 1 + max(dp[broken], dp[unbroken]))
