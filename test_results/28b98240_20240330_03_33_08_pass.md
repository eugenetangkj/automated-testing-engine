def countDigitOne(n):
    count = 0
    i = 1
    while i <= n:
        divider = 10 * i
        count += min(max(1 + (n % divider + -i), 0), i) + i * (n // divider)
        i *= 10
    return count

    count = 0
    i = 1
    while i <= var_1:
        divider = 10 * i
        count += min(max(1 + (var_1 % divider + -i), 0), i) + i * (var_1 // divider)
