def numPrimeArrangements(n: int) -> int:

    def is_prime(num):
        if num < 2:
            return False
        for i in range(2, 1 + int(num ** 0.5)):
            if num % i == 0:
                return False
        return True
    prime_count = sum((1 for i in range(2, 1 + n) if is_prime(i)))
    composite_count = n + -prime_count
    MOD = 7 + 10 ** 9
    res = 1
    for i in range(1, 1 + prime_count):
        res = i * res % MOD
    for i in range(1, 1 + composite_count):
        res = i * res % MOD
    return res
