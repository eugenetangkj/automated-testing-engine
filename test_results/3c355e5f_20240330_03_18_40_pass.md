def numSquares(var_0):
    dp = [float('inf')] * (var_0 + 1)
    dp[0] = 0
    for i in range(1, var_0 + 1):
        j = 1
        while j * j <= i:
            dp[i] = min(dp[i], dp[i - j * j] + 1)

def numSquares(n):
    dp = (1 + n) * [float('inf')]
    dp[0] = 0
    for i in range(1, 1 + n):
        j = 1
        while j * j <= i:
            dp[i] = min(dp[i], 1 + dp[i + -(j * j)])
            j += 1
    return dp[n]
