def num_dup_digits_at_most_n(var_0):
    count = 0
    for i in range(1, var_0 + 1):
        digits = set()
        num = i
        while num != 0:
            digit = num % 10
            if digit in digits:
                count += 1
                break
            else:
                digits.add(digit)
            num //= 10
    return count

def num_dup_digits_at_most_n(n):
    count = 0
    for i in range(1, 1 + n):
        digits = set()
        num = i
        while num != 0:
            digit = num % 10
            if digit in digits:
                count += 1
                break
            else:
                digits.add(digit)
            num //= 10
    return count

def num_dup_digits_at_most_n(var_1):
    count = 0
    for i in range(1, 1 + var_1):
        digits = set()
        num = i
        while num != 0:
            digit = num % 10
            if digit in digits:
                count += 1
                break
            else:
                digits.add(digit)
            num //= 10
    return count
