def findPaths(m, n, maxMove, startRow, startColumn):               
    MOD = 10**9 + 7                     
    dp = [[[0]* n for _ in range(m)] for _ in range(maxMove + 1)]
    moves = ((-1, 0), (1, 0), (0, -1), (0, 1))

    for k in range(1, maxMove + 1):
